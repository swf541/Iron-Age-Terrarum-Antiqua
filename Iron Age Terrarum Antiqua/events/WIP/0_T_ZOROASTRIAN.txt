# Zoroastrian events

namespace = dkzoroastrian

#####Sedreh Pushi ritual - child's event
character_event = {
	id = dkzoroastrian.1
	desc = "dkzoroastrian.1.desc"
	picture = "GFX_evt_child_bully"
	
	min_age = 6
	max_age = 16

	trigger = {
		or = {
			religion = zoroastrian_main
			religion = zurvanism
		}
		father = { is_alive = yes }
		NOT = {
				has_character_flag = had_sedreh_pushi
			}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "dkzoroastrian.11.name"	
		set_character_flag = had_sedreh_pushi
		hidden_tooltip = {
			father = {
				character_event = {
					id = dkzoroastrian.2
					days = 3
				}
			}
		}
	}
}
