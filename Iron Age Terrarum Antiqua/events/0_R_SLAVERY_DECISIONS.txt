# Character is notified he is becoming a slave
character_event = {
	id = slavery.1
	desc = slavery.1.desc
	picture = GFX_evt_child_reading # Placeholder
	
	is_triggered_only = yes # Triggered by decision: enslave
	
	option = {
		name = slavery.1.opta
		
		if = {
			limit = {
				is_ruler = yes
			}
			current_heir = {
				character_event = {
					id = slavery.2
				}
			}
			abdicate = yes # Slaves can't rule
		}
		add_trait = slave
		move_character = FROM
	}
}

# Heir is notified
